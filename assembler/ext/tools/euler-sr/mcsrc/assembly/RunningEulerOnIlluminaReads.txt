
This assumes reads are in reads.fasta.

countSpectrum reads.fasta reads.fasta.spect
sortTupleList reads.fasta.spect reads.fasta.sorted
fixErrorsVoting reads.fasta reads.fasta.sorted 20 reads.fasta.fixed -discardsFile reads.fasta.discards -minMult 2 -minVotes 2 -trim -maxTrim 5
assemble.pl reads.fasta.fixed -vertexSize 27


# The following step *may* work at once, otherwise you'll				
# have to run it in several steps.  There is a problem with memory
# fragmentation that we're currently working on.

simplifyGraph reads.fasta.fixed reads.fasta.simple -vertexSize 27 -minEdgeLength 60 -removeBulges 80 -removeLowCoverage 3.5


# -or-, if this runs out of memory

simplifyGraph reads.fasta.fixed reads.fasta.simple.1 -vertexSize 27 -minEdgeLength 60

simplifyGraph reads.fasta.simple.1 reads.fasta.simple.2 -vertexSize 27 -removeBulges 80

simplifyGraph reads.fasta.simple.2 reads.fasta.simple.3 -vertexSize 27 -removeLowCoverage 3.5
